JS Introduction

JavaScript is the programming language of the web. It runs in the browser and makes pages interactive.

document.getElementById("demo").innerHTML = "Hello World!";
alert("Hello!");
console.log("Debug message");
  • JS can change HTML content, style, and attributes
  • JS can react to user events (clicks, input, scroll)
  • JS runs in every modern browser — no install needed
  • JS can also run on servers (Node.js)

JS Where To

<!-- Inline event -->
<button onclick="alert('Hi')">Click</button>

<!-- Internal script (in <body> or <head>) -->
<script>
  console.log("Hello");
</script>

<!-- External script (recommended) -->
<script src="app.js" defer></script>
  • Place scripts before </body> or use defer to avoid blocking HTML parsing
  • defer — runs after HTML is parsed, maintains order
  • async — runs as soon as downloaded, order not guaranteed

[...a]             // spread copy

// Other

a.reverse()        // [3,2,1] (mutates)
a.flat()           // flatten nested arrays

JS Array Sort

JS Array Iteration

JS this Keyword

// In a method — refers to the object
const user = {
  name: "Alice",
  greet() { return this.name; }  // "Alice"
};

// In a regular function — refers to global (window) or undefined (strict)
function show() { console.log(this); }

// Arrow functions — inherit this from surrounding scope
const obj = {
  name: "Alice",
  greet: () => this.name  // wrong! arrow has no own this
  hello() {
    const fn = () => this.name;  // correct — inherits obj's this
    return fn();
  }
};

JS Functions

// Declaration — hoisted
function greet(name) {
  return "Hello, " + name;
}

// Expression — not hoisted
const greet = function(name) {
  return "Hello, " + name;
};

// Default parameters
function greet(name = "World") {
  return `Hello, ${name}!`;
}

// Rest parameters
function sum(...nums) {
  return nums.reduce((a, b) => a + b, 0);
}

JS Arrow Functions

// Standard function
function double(x) { return x * 2; }

// Arrow function
const double = (x) => x * 2;
const double = x  => x * 2;     // single param, no parens needed
const log    = () => console.log("hi");  // no params

// Multi-line
const add = (a, b) => {
  const result = a + b;
  return result;
};
  • Arrow functions have no own this — they inherit from enclosing scope
  • Cannot be used as constructors (no new)
  • No arguments object — use rest params instead

JS Closures

A closure is a function that remembers variables from its outer scope even after that scope has finished.

function makeCounter() {
  let count = 0;
  return function() {
    count++;
    return count;
  };
}

const counter = makeCounter();
counter(); // 1
counter(); // 2
counter(); // 3
// count is private — not accessible from outside

JS Scope

// Global scope — accessible everywhere
let globalVar = "I'm global";

function myFunc() {
  // Local (function) scope
  let localVar = "I'm local";

  if (true) {
    // Block scope (let/const only)
    let blockVar = "I'm block-scoped";
    var funcVar  = "I'm function-scoped"; // leaks out of block!
  }
  console.log(funcVar);  // accessible
  // console.log(blockVar); // ReferenceError
}

JS Hoisting

// Function declarations are fully hoisted
greet(); // works!
function greet() { console.log("Hi"); }

// var declarations are hoisted (but not initialised)
console.log(x); // undefined (not error)
var x = 5;

// let / const — hoisted but NOT initialised (Temporal Dead Zone)
// console.log(y); // ReferenceError
let y = 5;

JS Loops

// For loop
for (let i = 0; i < 5; i++) {
  console.log(i); // 0 1 2 3 4
}

// For...in — iterates object KEYS
const obj = { a: 1, b: 2 };
for (const key in obj) {
  console.log(key, obj[key]); // a 1, b 2
}

// For...of — iterates iterable VALUES
const arr = [10, 20, 30];
for (const val of arr) {
  console.log(val); // 10 20 30
}

// While
let n = 0;
while (n < 5) { n++; }

// Do...while — runs at least once
do {
  n++;
} while (n < 5);

// Break and continue
for (let i = 0; i < 10; i++) {
  if (i === 3) continue; // skip 3
  if (i === 6) break;    // stop at 6
}

JS Classes

class Animal {
  constructor(name, sound) {
    this.name  = name;
    this.sound = sound;
  }

  speak() {
    return `${this.name} says ${this.sound}`;
  }

  static create(name) {
    return new Animal(name, "...");
  }
}

class Dog extends Animal {
  constructor(name) {
    super(name, "Woof");
  }
  fetch() { return `${this.name} fetches!`; }
}

const dog = new Dog("Rex");
dog.speak(); // "Rex says Woof"

JS Destructuring

JS Spread & Rest

// Spread (...) — expand iterable
const arr1 = [1, 2, 3];
const arr2 = [...arr1, 4, 5];      // [1,2,3,4,5]
const copy = [...arr1];            // shallow copy

const obj1 = { a: 1 };
const obj2 = { ...obj1, b: 2 };   // { a:1, b:2 }

Math.max(...arr1);  // 3

// Rest (...) — collect remaining into array
function sum(first, ...others) {
  return first + others.reduce((a,b) => a+b, 0);
}

JS Promises

A Promise represents the eventual result of an asynchronous operation.

const p = new Promise((resolve, reject) => {
  setTimeout(() => resolve("Done!"), 1000);
});

p.then(result => console.log(result))  // "Done!"
 .catch(err   => console.error(err))
 .finally(()  => console.log("Always runs"));

// Promise.all — wait for multiple
Promise.all([fetch(url1), fetch(url2)])
  .then(([res1, res2]) => { /* both done */ });

JS Async / Await

Async/await is syntactic sugar over Promises — makes async code look synchronous.

async function getData() {
  try {
    const response = await fetch("https://api.example.com/data");
    const data     = await response.json();
    console.log(data);
  } catch (error) {
    console.error("Error:", error);
  }
}

// Parallel with Promise.all
async function getAll() {
  const [users, posts] = await Promise.all([
    fetch("/users").then(r => r.json()),
    fetch("/posts").then(r => r.json()),
  ]);
}

JS JSON

JSON (JavaScript Object Notation) is a text format for storing and transporting data.

// JS object → JSON string
const json = JSON.stringify({ name: "Alice", age: 25 });
// '{"name":"Alice","age":25}'

JSON.stringify(obj, null, 2);  // pretty-print with 2-space indent

// JSON string → JS object
const obj = JSON.parse('{"name":"Alice","age":25}');

// Deep clone (simple objects only)
const clone = JSON.parse(JSON.stringify(original));
  • JSON keys must be double-quoted strings
  • No functions, undefined, or comments allowed in JSON
